    <script>
        (function() {
            // Правильные ответы (значения option)
            const correctMap = {
                q1: 'B',   // предмет: конкретный перечень (5 принтеров + приложение)
                q2: 'A',   // 15 тыс. (разовая сумма, не ежемесячный оклад)
                q3: 'B',   // оплата после акта
                q4: 'A',   // чек не позднее следующего дня
                q5: 'B',   // материалы и способы – подрядчик, без режима работы
                q6: 'A',   // разовая работа или неодинаковый объём
                q7: 'A',   // акт с конкретными работами
                q8: 'A'    // НПД + обязанность сообщить об утрате права
            };

            const selectors = {
                q1: document.getElementById('sel1'),
                q2: document.getElementById('sel2'),
                q3: document.getElementById('sel3'),
                q4: document.getElementById('sel4'),
                q5: document.getElementById('sel5'),
                q6: document.getElementById('sel6'),
                q7: document.getElementById('sel7'),
                q8: document.getElementById('sel8')
            };

            const checkBtn = document.getElementById('checkBtn');
            const scoreDisplay = document.getElementById('scoreDisplay');
            const messageDisplay = document.getElementById('messageDisplay');

            // вспомогательная функция для сброса стилей
            function resetStyles() {
                for (let key in selectors) {
                    const sel = selectors[key];
                    sel.classList.remove('correct', 'wrong');
                }
            }

            // проверка
            function checkAll() {
                resetStyles();
                let correctCount = 0;
                let allSelected = true;

                for (let key in correctMap) {
                    const sel = selectors[key];
                    const selected = sel.value;
                    if (selected === '') {
                        allSelected = false;
                        continue;
                    }
                    const isCorrect = (selected === correctMap[key]);
                    if (isCorrect) {
                        sel.classList.add('correct');
                        correctCount++;
                    } else {
                        sel.classList.add('wrong');
                    }
                }

                // обновляем счёт
                scoreDisplay.textContent = `${correctCount} / 8`;

                // сообщение
                if (!allSelected) {
                    messageDisplay.innerHTML = '⚠️ Некоторые пункты не выбраны. Выберите все варианты.';
                    return;
                }

                if (correctCount === 8) {
                    messageDisplay.innerHTML = '✅ <span class="green">Отлично!</span> Все формулировки выбраны верно.';
                } else {
                    const wrong = 8 - correctCount;
                    messageDisplay.innerHTML = `❌ <span class="red">${wrong} ошибок</span>. Отметьте красные поля и выберите безопасные варианты.`;
                }
            }

            // событие на кнопку
            checkBtn.addEventListener('click', checkAll);

            // также можно сделать автоматическую проверку, но по заданию — только по кнопке.
            // дополнительно: при изменении селекта можно убирать подсветку, чтобы не сбивало.
            for (let key in selectors) {
                selectors[key].addEventListener('change', function() {
                    this.classList.remove('correct', 'wrong');
                    // обновим счёт, но только если все выбраны — по желанию, но оставим только по кнопке
                });
            }

            // инициализация: сообщение
            messageDisplay.innerHTML = 'Выберите варианты и нажмите «Проверить»';
            scoreDisplay.textContent = '0 / 8';
        })();
    </script>
